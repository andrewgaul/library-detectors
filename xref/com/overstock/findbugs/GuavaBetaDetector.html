27    private HashMap<FieldOrMethod, Boolean> memberBetaAnnotated = new HashMap<FieldOrMethod, Boolean>();

29    @Override
30    public void sawOpcode(int seen) {
31      switch (seen) {
32        case INVOKEINTERFACE:
33        case INVOKESTATIC:
34        case INVOKESPECIAL:
35        case INVOKEVIRTUAL: {
36          JavaClass javaClass = getOperandLibraryClass();
37          if (javaClass != null) {
38            checkJavaClass(javaClass);
39            Method method = getMethod(javaClass, getNameConstantOperand(), getSigConstantOperand());
40            if (method != null && isBetaAnnotated(method)) {
41              bugReporter.reportBug(bugInstance("GBU_GUAVA_BETA_METHOD_USAGE").addCalledMethod(this));
42            }
43          }
44          break;
45        }
46        case GETFIELD:
47        case GETSTATIC: {
48          JavaClass javaClass = getOperandLibraryClass();
49          if (javaClass != null) {
50            checkJavaClass(javaClass);
51            Field field = getField(javaClass, getNameConstantOperand());
52            if (field != null && isBetaAnnotated(field)) {
53              bugReporter.reportBug(
54                bugInstance("GBU_GUAVA_BETA_FIELD_USAGE").addReferencedField(this));
55            }
56          }
57          break;
58        }
59      }
60    }

62    /**
63     * @return the JavaClass for the current operand, provided that it is not an application class;
64     * null otherwise.
65     */
66    private JavaClass getOperandLibraryClass() {
67      try {
68        final JavaClass javaClass = Repository.lookupClass(getClassConstantOperand());
69        return AnalysisContext.currentAnalysisContext().isApplicationClass(javaClass)
70            ? null
71            : javaClass;
72      }
73      catch (ClassNotFoundException e) {
74        bugReporter.reportMissingClass(e);
75        return null;
76      }
77  
78    }

80    private Method getMethod(JavaClass javaClass, String name, String sig) {
81      for (JavaClass currentClass = javaClass; currentClass != null; currentClass = getSuperclass(currentClass)) {
82        for (Method method : currentClass.getMethods()) {
83          if (name.equals(method.getName()) && sig.equals(method.getSignature())) {
84            return method;
85          }
86        }
87      }
88      if (!javaClass.isAbstract()) {
89        bugReporter.logError("Unable to locate method " + javaClass.getClassName() + "." + name + sig);
90      }
91      return null;
92    }

94    private JavaClass getSuperclass(JavaClass javaClass) {
95      try {
96        return javaClass.getSuperClass();
97      }
98      catch (ClassNotFoundException e) {
99        bugReporter.reportMissingClass(e);
100       return null;
101     }
102   }
103 
104   private Field getField(JavaClass javaClass, String name) {
105     for (JavaClass currentClass = javaClass; currentClass != null; currentClass = getSuperclass(currentClass)) {
106       for (Field field : currentClass.getFields()) {
107         if (name.equals(field.getName())) {
108           return field;
109         }
110       }
111     }
112     bugReporter.logError("Unable to locate field " + javaClass.getClassName() + "." + name);
113     return null;
114   }
115 
116   private void checkJavaClass(JavaClass javaClass) {
117     if (isBetaAnnotated(javaClass.getAnnotationEntries())) {
118       bugReporter.reportBug(bugInstance("GBU_GUAVA_BETA_CLASS_USAGE").addClass(javaClass));
119     }
120   }

122   private BugInstance bugInstance(String type) {
123     return new BugInstance(this, type, NORMAL_PRIORITY).addClassAndMethod(this).addSourceLine(this);
124   }

126   private boolean isBetaAnnotated(FieldOrMethod annotated) {
127     Boolean result = memberBetaAnnotated.get(annotated);
128     if (result != null) {
129       return result;
130     }
131     else {
132       result = isBetaAnnotated(annotated.getAnnotationEntries());
133       memberBetaAnnotated.put(annotated, result);
134       return result;
135     }
136   }
137 
138   private boolean isBetaAnnotated(AnnotationEntry[] annotationEntries) {
139     for (AnnotationEntry annotationEntry : annotationEntries) {
140       if (BETA.equals(annotationEntry.getAnnotationType())) {
141         return true;
142       }
143     }
144     return false;
145   }
